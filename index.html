Carnation Points

“Responsive States — Know exactly where your layout stands.”

Handling Window Resize

Example: none

Define carnationPoints()

        
        const points = carnationPoints(".myBox");

Create Breakpoints

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
        });

Get Breakpoint on Init

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
          onInit(point) {
            console.log("Init:", point.breakpoint);
          },
        });

Get Breakpoint on Resize

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
          onChange(point) {
            console.log("Resize:", point.breakpoint);
          },
        });

Add new breakpoints

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
        });

        points.addPoint([
          { name: "lg", min: 1200 },
          { name: "xl", min: 1600 },
        ]);

Destroy breakpoints

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
        });

        points.destroy();

Get Breakpoints

        
        const points = carnationPoints(".myBox", {
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
        });

        points.getBreakpoint()
        if(points.getBreakpoint() === "mobile"){
          console.log("You are in mobile breakpoint");
        }

Different way usage

        
        const points = carnationPoints("#myBox", {
          debug: true,
          breakpoints: [
            { name: "mobile", max: 767 },
            { name: "tablet", min: 768, max: 1023 },
            { name: "desktop", min: 1024 },
          ],
          onInit(point) {
            if (point.breakpoint) {
              document.querySelector(
                "#myBox"
              ).textContent = `Example: ${point.breakpoint}`;
            }
          },
          onChange(point) {
            document.querySelector(
              "#myBox"
            ).textContent = `Example: ${point.breakpoint}`;

            if (points.getBreakpoint() === "mobile") {
              console.log("You are in mobile breakpoint");
            } else {
              console.log("You are not in mobile breakpoint");
            }
          },
        });
